1) Machine Learning Techniques for Predictive Analytics

# Experiment 1
# Machine Learning Techniques for Predictive Analytics

library(caret)
library(ggplot2)

# Dataset
data(iris)

# Split data
set.seed(100)
trainIndex <- createDataPartition(iris$Species, p=0.8, list=FALSE)

train <- iris[trainIndex, ]
test <- iris[-trainIndex, ]

# Train model
model <- train(Species ~ ., data=train, method="rpart")

# Prediction
pred <- predict(model, test)

# Result
confusionMatrix(pred, test$Species)

2) Predict Customer Credit Risk Using Linear Regression

# Experiment 2
# Customer Credit Risk

library(ggplot2)

# Data
income <- c(25000,30000,40000,50000,60000)
loan <- c(5000,7000,9000,11000,13000)

data <- data.frame(income, loan)

# Linear regression
model <- lm(loan ~ income, data=data)

# Summary
summary(model)

# Plot
ggplot(data, aes(income, loan)) +
  geom_point() +
  geom_smooth(method="lm")

6) Financial / Bank Default Prediction

# Experiment 6
# Bank Default Prediction

library(caret)

# Sample data
income <- c(25000,30000,40000,50000,60000)
loan <- c(5000,7000,9000,11000,13000)
status <- c("No","No","Yes","Yes","Yes")

data <- data.frame(income, loan, status)

# Convert factor
data$status <- as.factor(data$status)

# Model
model <- train(status ~ income + loan,
               data=data,
               method="rpart")

# Prediction
predict(model)
